37float GetLineAngle(cv::Vec4f line) // 起点xy 终点xy
38{
39
40 cv::Point p1(line[0], line[1]);
41 cv::Point p2(line[2], line[3]);
42
43 float angle = 0;
44
45 float angle_ = atan2(p1.y - p2.y, p1.x - p2.x);
46 angle = (angle_ * 180) / CV_PI;
47 if (angle_ >= -CV_PI && angle_ < 0)
48 {
49 angle = 180 + (180 + angle);
50 }
51
52 return angle;
53}

112 void ImageProcess(cv::Mat src)
113 {
114 // color = this->color;
115 std::vector<cv::Mat> channels;
116 cv::Mat mat;
117 cv::Mat channel_img;
118 cv::Mat threshold_img;
119 cv::Mat black_img = cv::Mat::zeros(src.size(), CV_8UC1);
120 // 三通道分离
121 cv::split(src, channels);
122
123 // 减去干扰色
124 if (color == 2) // 红色 2
125 {
126 cv::subtract(channels[2], channels[0], channel_img);
127 }
128 else if (color == 3) // 蓝色 3
129 {
130 cv::subtract(channels[0], channels[2], channel_img);
131 }
132
133 // 选择目标颜色进行二值化
134 cv::threshold(channel_img, threshold_img, 100, 255, cv::THRESH_BINARY);
135 // 开操作
136 cv::Mat element = cv::getStructuringElement(cv::MORPH_RECT, cv::Size(3, 3));
137 cv::morphologyEx(threshold_img, mat, cv::MORPH_OPEN, element);
138 // 寻找轮廓
139 std::vector<std::vector<cv::Point>> contours;
140 std::vector<cv::Vec4i> hierarchy;
141 cv::findContours(mat, contours, hierarchy, cv::RETR_EXTERNAL, cv::CHAIN_APPROX_SIMPLE);
142
143 for (auto it = contours.begin(); it != contours.end();)
144 {
145 double area = contourArea(*it);
146 if (area < minArea)
147 {
148 it = contours.erase(it);
149 }
150 else
151 {
152 ++it;
153 }
154 }
155 cv::drawContours(black_img, contours, -1, cv::Scalar(255, 255, 255), 3, 4);
156 // 凸包检测
157 std::vector<std::vector<cv::Point>> hull(contours.size());
158 for (int i = 0; i < contours.size(); i++)
159 {
160 // 寻找凸包
161 cv::convexHull(cv::Mat(contours[i]), hull[i], false);
162 }
163 for (int i = 0; i < contours.size(); i++)
164 {
165 cv::fillConvexPoly(black_img, hull[i], cv::Scalar(255, 255, 255), cv::LINE_AA, 0);
166 }
167 std::vector<std::vector<cv::Point>> contours_;
168 std::vector<cv::Vec4i> hierarchy_;
169 cv::findContours(black_img, contours_, hierarchy_, cv::RETR_EXTERNAL, cv::CHAIN_APPROX_SIMPLE);
170 cv::Mat result_img = cv::Mat::zeros(src.size(), CV_8UC3);
171 cv::drawContours(result_img, contours_, -1, cv::Scalar(255, 255, 255), 3, 4);
172 processed_img = result_img; // 将处理过的图像传回给类实例
173 }
